is_republican_state = {
	OR = {
		AND = {
			check_variable = { republican_support > democratic_support }
		}

		AND = {
			check_variable = { republican_support = democratic_support }
		}
	}
}

is_democratic_state = {
	OR = {
		AND = {
			check_variable = { democratic_support > republican_support }
		}

		AND = {
			check_variable = { democratic_support = republican_support }
		}
	}
}
