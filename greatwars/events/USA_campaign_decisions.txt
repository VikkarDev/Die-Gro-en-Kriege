# texas campaign run successful
country_event = {
    id = USA_campaign_decisions.1
    title = USA_campaign_decisions.1.t
    desc = USA_campaign_decisions.1.d 
    picture = GFX_report_event_generic_rally

    #fire_only_once = yes 
    is_triggered_only = yes

    option = {
        name = USA_campaign_decisions.1.a 

		if = {
			limit = {
				has_government = traditional_conservatism
			}
			375 = { add_to_variable = { republican_support = 3 } }
			375 = { subtract_from_variable = { democratic_support = 3 } }
			USA_modify_republican_campaign_running_effect = yes
			custom_effect_tooltip = USA_increase_republican_support_three
		}
		else_if = {
			limit = {
				has_government = centrism
			}
			375 = { add_to_variable = { democratic_support = 3 } }
			878 = { subtract_from_variable = { republican_support = 3 } }
			USA_modify_democratic_campaign_running_effect = yes
			custom_effect_tooltip = USA_increase_democratic_support_three
		}

    }
}
