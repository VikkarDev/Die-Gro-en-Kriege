add_namespace = britain_naval

# NAVAL ARMS RACE
country_event = {
	id = britain_naval.1
	title = britain_naval.1.t
	desc = britain_naval.1.d
	
	is_triggered_only = yes

	option = {
		name = britain_naval.1.a # John Fisher
		add_political_power = 30
		hidden_effect = {
			navy_experience = 30
			add_tech_bonus = {
				bonus = 0.2
				uses = 3
				category = naval_equipment
			}
			set_variable = { ENG_first_sea_lord = 2 }
		}
	}

	option = {
		name = britain_naval.1.b # Arthur Wilson
		add_political_power = 30
		hidden_effect = {
			add_offsite_building = {
				type = naval_factory
				level = 3
			}
			set_variable = { ENG_first_sea_lord = 3 }
		}
	}
	option = {
		name = britain_naval.1.c # Francis Bridgeman
		add_political_power = 30
		hidden_effect = {
			add_war_support = 0.15
			research_speed_factor = -0.02
			amphibious_invasion = -0.03
			convoy_raiding_efficiency_factor = -0.02
			experience_gain_navy_factor = -0.2
		}
		set_variable = { ENG_first_sea_lord = 4 }
	}
}
